import { useState, useRef, useEffect } from "react";

const SYSTEM_PROMPT = `Tu es RIAL Advisor, l'assistant financier IA de référence pour l'Afrique, créé par RIAL GROUP (Cameroun).

Ton rôle :
- Conseiller financier personnel accessible à tous les Africains
- Expert en finance africaine : OHADA, SYSCOHADA, BEAC, BCEAO, COBAC, Mobile Money
- Spécialiste des corridors de paiement : FCFA, NGN, KES, GHS, XOF, ZAR
- Conseils en épargne, budget, crédit, investissement, transferts d'argent
- Finance islamique (Mourabaha, Ijara, Sukuk) pour les populations musulmanes

Langues : Tu détectes automatiquement la langue de l'utilisateur et réponds dans la même langue (Français, Anglais, et si possible Swahili ou Haoussa).

Ton style :
- Chaleureux, accessible, professionnel
- Explications simples pour les non-bancarisés
- Jamais de jargon inutile
- Toujours pratique et actionnable
- Tu connais les réalités terrain africaines (Mobile Money, tontines, marchés informels)

Tu t'appelles RIAL Advisor. Tu es fier d'être africain et de servir l'Afrique.`;

const SUGGESTIONS = [
  { fr: "Comment épargner avec Mobile Money ?", en: "How to save with Mobile Money?" },
  { fr: "Expliquez-moi la finance islamique", en: "Explain Islamic finance to me" },
  { fr: "Comment envoyer de l'argent au Nigeria ?", en: "How to send money to Nigeria?" },
  { fr: "Aidez-moi à faire un budget", en: "Help me make a budget" },
];

const AfricaIcon = () => (
  <svg width="22" height="22" viewBox="0 0 24 24" fill="none">
    <path d="M12 2C6.48 2 2 6.48 2 12s4.48 10 10 10 10-4.48 10-10S17.52 2 12 2z" fill="#C8A96E" opacity="0.2"/>
    <path d="M12 2C6.48 2 2 6.48 2 12s4.48 10 10 10 10-4.48 10-10S17.52 2 12 2z" stroke="#C8A96E" strokeWidth="1.5"/>
    <path d="M8 7l2 3-1 2 2 1 1 3 3-1 1-2-2-3 1-2-3-1z" fill="#C8A96E" opacity="0.6"/>
  </svg>
);

const SendIcon = () => (
  <svg width="20" height="20" viewBox="0 0 24 24" fill="none">
    <path d="M22 2L11 13M22 2L15 22l-4-9-9-4 20-7z" stroke="currentColor" strokeWidth="2" strokeLinecap="round" strokeLinejoin="round"/>
  </svg>
);

const LoadingDots = () => (
  <div style={{ display: "flex", gap: 5, alignItems: "center", padding: "4px 0" }}>
    {[0, 1, 2].map(i => (
      <div key={i} style={{
        width: 7, height: 7, borderRadius: "50%", background: "#C8A96E",
        animation: "bounce 1.2s infinite",
        animationDelay: `${i * 0.2}s`
      }} />
    ))}
  </div>
);

export default function RIALAdvisor() {
  const [messages, setMessages] = useState([
    {
      role: "assistant",
      content: "Bonjour ! Je suis **RIAL Advisor**, votre conseiller financier IA pour l'Afrique 🌍\n\nJe peux vous aider avec :\n• Épargne & budget\n• Transferts d'argent\n• Crédit & investissement\n• Finance islamique\n• Mobile Money\n\nComment puis-je vous aider aujourd'hui ?"
    }
  ]);
  const [input, setInput] = useState("");
  const [loading, setLoading] = useState(false);
  const [showSuggestions, setShowSuggestions] = useState(true);
  const messagesEndRef = useRef(null);
  const inputRef = useRef(null);

  useEffect(() => {
    messagesEndRef.current?.scrollIntoView({ behavior: "smooth" });
  }, [messages, loading]);

  const formatMessage = (text) => {
    return text
      .replace(/\*\*(.*?)\*\*/g, '<strong>$1</strong>')
      .replace(/\*(.*?)\*/g, '<em>$1</em>')
      .replace(/\n/g, '<br/>');
  };

  const sendMessage = async (text) => {
    const userText = text || input.trim();
    if (!userText) return;

    setInput("");
    setShowSuggestions(false);
    setLoading(true);

    const newMessages = [...messages, { role: "user", content: userText }];
    setMessages(newMessages);

    const apiMessages = newMessages.map(m => ({ role: m.role, content: m.content }));

    try {
      const response = await fetch("https://api.anthropic.com/v1/messages", {
        method: "POST",
        headers: { "Content-Type": "application/json" },
        body: JSON.stringify({
          model: "claude-sonnet-4-6",
          max_tokens: 1000,
          system: SYSTEM_PROMPT,
          messages: apiMessages
        })
      });

      const data = await response.json();
      const reply = data.content?.[0]?.text || "Désolé, une erreur s'est produite. Réessayez.";

      setMessages(prev => [...prev, { role: "assistant", content: reply }]);
    } catch {
      setMessages(prev => [...prev, {
        role: "assistant",
        content: "Connexion momentanément indisponible. Veuillez réessayer."
      }]);
    } finally {
      setLoading(false);
    }
  };

  const handleKey = (e) => {
    if (e.key === "Enter" && !e.shiftKey) {
      e.preventDefault();
      sendMessage();
    }
  };

  return (
    <div style={{
      minHeight: "100vh",
      background: "#0A0A0F",
      display: "flex",
      flexDirection: "column",
      alignItems: "center",
      fontFamily: "'Inter', -apple-system, sans-serif",
    }}>
      <style>{`
        @import url('https://fonts.googleapis.com/css2?family=Inter:wght@300;400;500;600&family=Space+Grotesk:wght@500;600;700&display=swap');
        * { box-sizing: border-box; margin: 0; padding: 0; }
        ::-webkit-scrollbar { width: 4px; }
        ::-webkit-scrollbar-track { background: transparent; }
        ::-webkit-scrollbar-thumb { background: #2a2a3a; border-radius: 2px; }
        @keyframes bounce {
          0%, 60%, 100% { transform: translateY(0); }
          30% { transform: translateY(-6px); }
        }
        @keyframes fadeIn {
          from { opacity: 0; transform: translateY(8px); }
          to { opacity: 1; transform: translateY(0); }
        }
        @keyframes shimmer {
          0% { background-position: -200% center; }
          100% { background-position: 200% center; }
        }
        .msg-in { animation: fadeIn 0.3s ease; }
        .suggestion-btn:hover {
          background: rgba(200, 169, 110, 0.12) !important;
          border-color: rgba(200, 169, 110, 0.5) !important;
          transform: translateY(-1px);
        }
        .send-btn:hover:not(:disabled) {
          background: #D4B97E !important;
          transform: scale(1.05);
        }
        .send-btn:disabled { opacity: 0.4; cursor: not-allowed; }
        textarea:focus { outline: none; border-color: rgba(200, 169, 110, 0.4) !important; }
        textarea { resize: none; }
      `}</style>

      {/* Header */}
      <div style={{
        width: "100%",
        maxWidth: 780,
        padding: "20px 24px 16px",
        borderBottom: "1px solid rgba(255,255,255,0.06)",
        display: "flex",
        alignItems: "center",
        gap: 14,
        position: "sticky",
        top: 0,
        background: "rgba(10,10,15,0.95)",
        backdropFilter: "blur(12px)",
        zIndex: 10,
      }}>
        {/* Logo */}
        <div style={{
          width: 42, height: 42,
          background: "linear-gradient(135deg, #1a1a28, #252535)",
          border: "1px solid rgba(200,169,110,0.3)",
          borderRadius: 10,
          display: "flex", alignItems: "center", justifyContent: "center",
          flexShrink: 0,
        }}>
          <AfricaIcon />
        </div>

        <div style={{ flex: 1 }}>
          <div style={{
            fontFamily: "'Space Grotesk', sans-serif",
            fontSize: 17,
            fontWeight: 700,
            background: "linear-gradient(90deg, #C8A96E, #E8C98E, #C8A96E)",
            backgroundSize: "200% auto",
            WebkitBackgroundClip: "text",
            WebkitTextFillColor: "transparent",
            animation: "shimmer 3s linear infinite",
          }}>
            RIAL Advisor
          </div>
          <div style={{ fontSize: 11, color: "#5a5a7a", marginTop: 1, letterSpacing: "0.04em" }}>
            L'IA Financière de l'Afrique · by RIAL GROUP
          </div>
        </div>

        <div style={{
          display: "flex", alignItems: "center", gap: 6,
          background: "rgba(34, 197, 94, 0.08)",
          border: "1px solid rgba(34, 197, 94, 0.2)",
          borderRadius: 20, padding: "4px 10px",
        }}>
          <div style={{ width: 6, height: 6, borderRadius: "50%", background: "#22c55e" }} />
          <span style={{ fontSize: 11, color: "#22c55e", fontWeight: 500 }}>En ligne</span>
        </div>
      </div>

      {/* Messages */}
      <div style={{
        flex: 1, width: "100%", maxWidth: 780,
        overflowY: "auto", padding: "24px 24px 0",
        display: "flex", flexDirection: "column", gap: 16,
      }}>
        {messages.map((msg, i) => (
          <div key={i} className="msg-in" style={{
            display: "flex",
            flexDirection: msg.role === "user" ? "row-reverse" : "row",
            alignItems: "flex-end", gap: 10,
          }}>
            {msg.role === "assistant" && (
              <div style={{
                width: 30, height: 30, borderRadius: 8, flexShrink: 0,
                background: "linear-gradient(135deg, #1a1a28, #252535)",
                border: "1px solid rgba(200,169,110,0.25)",
                display: "flex", alignItems: "center", justifyContent: "center",
              }}>
                <AfricaIcon />
              </div>
            )}

            <div style={{
              maxWidth: "75%",
              background: msg.role === "user"
                ? "linear-gradient(135deg, #C8A96E, #B8965E)"
                : "rgba(255,255,255,0.04)",
              border: msg.role === "user" ? "none" : "1px solid rgba(255,255,255,0.07)",
              borderRadius: msg.role === "user" ? "16px 16px 4px 16px" : "4px 16px 16px 16px",
              padding: "12px 16px",
              fontSize: 14,
              lineHeight: 1.65,
              color: msg.role === "user" ? "#0A0A0F" : "#d4d4e8",
              fontWeight: msg.role === "user" ? 500 : 400,
            }}
              dangerouslySetInnerHTML={{ __html: formatMessage(msg.content) }}
            />
          </div>
        ))}

        {loading && (
          <div className="msg-in" style={{ display: "flex", alignItems: "flex-end", gap: 10 }}>
            <div style={{
              width: 30, height: 30, borderRadius: 8, flexShrink: 0,
              background: "linear-gradient(135deg, #1a1a28, #252535)",
              border: "1px solid rgba(200,169,110,0.25)",
              display: "flex", alignItems: "center", justifyContent: "center",
            }}>
              <AfricaIcon />
            </div>
            <div style={{
              background: "rgba(255,255,255,0.04)",
              border: "1px solid rgba(255,255,255,0.07)",
              borderRadius: "4px 16px 16px 16px",
              padding: "12px 16px",
            }}>
              <LoadingDots />
            </div>
          </div>
        )}

        {/* Suggestions */}
        {showSuggestions && messages.length === 1 && (
          <div style={{ marginTop: 8 }}>
            <p style={{ fontSize: 11, color: "#5a5a7a", marginBottom: 10, letterSpacing: "0.04em" }}>
              QUESTIONS FRÉQUENTES
            </p>
            <div style={{ display: "flex", flexWrap: "wrap", gap: 8 }}>
              {SUGGESTIONS.map((s, i) => (
                <button key={i} className="suggestion-btn"
                  onClick={() => sendMessage(s.fr)}
                  style={{
                    background: "rgba(255,255,255,0.03)",
                    border: "1px solid rgba(255,255,255,0.09)",
                    borderRadius: 20, padding: "8px 14px",
                    color: "#a0a0c0", fontSize: 12, cursor: "pointer",
                    transition: "all 0.2s ease", fontFamily: "'Inter', sans-serif",
                  }}>
                  {s.fr}
                </button>
              ))}
            </div>
          </div>
        )}

        <div ref={messagesEndRef} style={{ height: 16 }} />
      </div>

      {/* Input */}
      <div style={{
        width: "100%", maxWidth: 780,
        padding: "16px 24px 24px",
        position: "sticky", bottom: 0,
        background: "rgba(10,10,15,0.95)",
        backdropFilter: "blur(12px)",
      }}>
        <div style={{
          display: "flex", gap: 10, alignItems: "flex-end",
          background: "rgba(255,255,255,0.04)",
          border: "1px solid rgba(255,255,255,0.09)",
          borderRadius: 16, padding: "10px 10px 10px 16px",
          transition: "border-color 0.2s",
        }}>
          <textarea
            ref={inputRef}
            value={input}
            onChange={e => setInput(e.target.value)}
            onKeyDown={handleKey}
            placeholder="Posez votre question financière... / Ask your financial question..."
            rows={1}
            style={{
              flex: 1, background: "transparent",
              border: "none", color: "#d4d4e8",
              fontSize: 14, lineHeight: 1.5,
              fontFamily: "'Inter', sans-serif",
              maxHeight: 120, overflowY: "auto",
            }}
            onInput={e => {
              e.target.style.height = "auto";
              e.target.style.height = Math.min(e.target.scrollHeight, 120) + "px";
            }}
          />
          <button className="send-btn"
            onClick={() => sendMessage()}
            disabled={!input.trim() || loading}
            style={{
              width: 38, height: 38, borderRadius: 10,
              background: "#C8A96E", border: "none",
              color: "#0A0A0F", cursor: "pointer",
              display: "flex", alignItems: "center", justifyContent: "center",
              transition: "all 0.2s ease", flexShrink: 0,
            }}>
            <SendIcon />
          </button>
        </div>
        <p style={{
          textAlign: "center", fontSize: 10, color: "#3a3a5a",
          marginTop: 10, letterSpacing: "0.03em",
        }}>
          RIAL Advisor · Conseil financier IA · RIAL GROUP Cameroun
        </p>
      </div>
    </div>
  );
}
